// ✅ PHANTOM STANDARD: Constant-time comparison — always
import { timingSafeEqual } from 'node:crypto'

function safeCompare(a: string, b: string): boolean {
  // Pad to same length to prevent early-exit on length mismatch
  const bufA = Buffer.from(a.padEnd(64))
  const bufB = Buffer.from(b.padEnd(64))
  return timingSafeEqual(bufA, bufB) && a.length === b.length
}

  type:        argon2.argon2id,  // resistant to GPU and side-channel
  memoryCost: 65536,            // 64 MB — tune to your server
  timeCost:   3,                // iterations
  parallelism: 4,               // threads
}

export const hashPassword = (plain: string) =>
  argon2.hash(plain, ARGON_OPTIONS)

export const verifyPassword = (hash: string, plain: string) =>
  argon2.verify(hash, plain)

Security Doctrine
01
Assume Breach
Design controls assuming the perimeter has already failed. Ask not only how to prevent — but how to detect and limit damage when it happens. Defense begins after the wall.
02
Never Trust, Always Verify
Every input is hostile until validated. Every caller is unauthenticated until proven. Every dependency is a supply chain risk until reviewed. Trust is explicit, scoped, and time-limited.
03
Defense in Depth
No single control is sufficient. Layer validation, authentication, encryption, monitoring, and rate-limiting. A breach that bypasses one control must still face five more.
04
Least Privilege Everywhere
Tokens, roles, service accounts, API keys, database users — every principal has exactly the permissions its task requires. No more. Over-permissioning is a vulnerability with a CVSS score.
05
Security is Observable
A breach you cannot detect is worse than one you can. Every control must have corresponding logging and alerting. Audit trails are tamper-evident. Anomalies surface before they become incidents.
The question is never whether you will be attacked.
The question is whether you will know.
